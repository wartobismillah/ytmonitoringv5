YT Monitor — Setup

Cukup sekali, ~10 menit

Ikuti langkah berikut:
1
Buka sheets.new → buat Google Sheet baru (nama bebas, biarkan kosong).
2
Di Google Sheet → klik Extensions → Apps Script → hapus semua kode yang ada → paste kode ini:
// ═══════════════════════════════════════════════════ // YT Monitor — Apps Script (Google Drive Storage + Sheets Backup) // Data utama disimpan sbg JSON di Drive (cepat utk sinkronisasi web app). // Setiap kali disimpan, otomatis JUGA ditulis ulang ke tab Sheet // "Backup Database" dlm bentuk kolom yang bisa dibaca manusia langsung. // ═══════════════════════════════════════════════════ var FILE_NAME = "ytmonitor-data.json"; var BACKUP_SHEET_NAME = "Backup Database"; function getOrCreateFile() { var files = DriveApp.getFilesByName(FILE_NAME); if (files.hasNext()) return files.next(); return DriveApp.createFile(FILE_NAME, '{"channels":[],"apiKeys":[]}', MimeType.PLAIN_TEXT); } function doGet(e) { try { var content = getOrCreateFile().getBlob().getDataAsString(); return ContentService .createTextOutput(content) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({error:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function doPost(e) { try { var data = e.postData.contents; var file = getOrCreateFile(); file.setContent(data); // Tulis juga ke tab Sheet "Backup Database" supaya data tetap bisa // dibaca manusia walau web app/file JSON error. Dibungkus try/catch // sendiri supaya kalau bagian ini gagal, sinkronisasi utama tidak ikut gagal. try { var parsed = JSON.parse(data); writeBackupSheet_(parsed.channels || []); } catch (sheetErr) { Logger.log("Backup sheet gagal ditulis: " + sheetErr.toString()); } return ContentService.createTextOutput('{"status":"ok"}') .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({error:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } // Tulis ulang seluruh tab "Backup Database" berdasarkan state channels saat ini. function writeBackupSheet_(channelList) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(BACKUP_SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(BACKUP_SHEET_NAME); } var headers = ["Channel Name", "URL/ID", "Stage", "Subscribers", "Total Videos", "Last Upload Date", "Average Views"]; var rows = channelList.map(function(ch) { var videos = ch.videos || []; var lastUploadDate = "—"; if (videos.length > 0) { var sorted = videos.slice().sort(function(a, b) { return new Date(b.publishedAt) - new Date(a.publishedAt); }); if (sorted[0] && sorted[0].publishedAt) { lastUploadDate = Utilities.formatDate(new Date(sorted[0].publishedAt), Session.getScriptTimeZone(), "yyyy-MM-dd HH:mm"); } } var avgViews = 0; if (videos.length > 0) { var totalViews = videos.reduce(function(sum, v) { return sum + (v.views || 0); }, 0); avgViews = Math.round(totalViews / videos.length); } return [ ch.name || ch.inputValue || "—", ch.inputValue || ch.ytId || "—", "Tahap " + (ch.stage || 1), ch.subscribers || 0, ch.videoCount || 0, lastUploadDate, avgViews ]; }); sheet.clearContents(); sheet.getRange(1, 1, 1, headers.length).setValues([headers]).setFontWeight("bold").setBackground("#f0f0f0"); sheet.setFrozenRows(1); if (rows.length > 0) { sheet.getRange(2, 1, rows.length, headers.length).setValues(rows); } sheet.autoResizeColumns(1, headers.length); }
Klik 💾 Save → klik ▶ Run → pilih doGet → izinkan akses Google. Setelah ini, tab baru bernama "Backup Database" akan otomatis muncul & terisi tiap kali ada perubahan data.
3
Klik Deploy → New deployment → pilih type Web app → isi:
Execute as: Me  ·  Who has access: Anyone → Deploy → izinkan → copy Web app URL.
4
Paste URL + buat PIN tim di form bawah, lalu klik Cek & Generate File. Kamu akan download file index.html baru → upload ke GitHub menggantikan file lama.
Dari Deploy → Manage deployments → copy URL-nya
Semua anggota tim pakai PIN yang sama untuk masuk

✓ Koneksi berhasil!

File index.html sudah di-generate dengan konfigurasi tertanam di dalamnya. Sekarang upload ke GitHub:

1Klik tombol Download di bawah → simpan sebagai index.html
2Buka repo GitHub → klik file index.html lama → klik ✏️ Edit → hapus semua → paste isi file baru → Commit
3Atau lebih mudah: hapus file lama → Upload file baru langsung
4Selesai! Tim buka link GitHub Pages → langsung masuk layar PIN
